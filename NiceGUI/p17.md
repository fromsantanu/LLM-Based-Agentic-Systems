# **NiceGUI Code Example**

```python
from nicegui import ui

ui.label('Form with Simple Validation').classes('text-h5')

# Text inputs
name = ui.input('Enter your name')
age = ui.input('Enter your age')

def validate_form():
    # Get values
    name_val = name.value
    age_val = age.value

    # Check 1: Name must not be empty
    if not name_val:
        ui.notify('Name cannot be empty', color='red')
        return

    # Check 2: Age must be a number
    if not age_val.isdigit():
        ui.notify('Age must be a number', color='red')
        return

    # Check 3: Age should be reasonable (optional)
    if int(age_val) < 1 or int(age_val) > 120:
        ui.notify('Age must be between 1 and 120', color='red')
        return

    # If all checks pass
    ui.notify(f'Success! Name: {name_val}, Age: {age_val}', color='green')

ui.button('Submit', on_click=validate_form)

ui.run()
```
