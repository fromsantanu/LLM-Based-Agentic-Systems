# **ğŸ’» Code Example (Simple and Clean)**

Save this file as: **chat.py**

```python
from nicegui import ui

# this list will store chat messages
messages = []

def send_message():
    user_text = input_box.value
    if user_text.strip() == "":
        return
    
    # add user's message to the list
    messages.append(f"You: {user_text}")
    
    # create a simple reply
    reply = f"Bot: You said '{user_text}'"
    messages.append(reply)
    
    # update the message area
    chat_area.clear()
    for msg in messages:
        ui.label(msg)
    
    # clear the input box
    input_box.value = ""

with ui.column().classes('w-1/2'):
    ui.label("Simple Chat Window")

    # area where messages will be displayed
    chat_area = ui.column().classes('border rounded p-4 h-64 overflow-y-auto bg-white')

    # input area
    with ui.row():
        input_box = ui.input(placeholder="Type your message here...")
        ui.button("Send", on_click=send_message)

ui.run()
```
