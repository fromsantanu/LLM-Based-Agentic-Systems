Create a new file:

Paste this code:

```python
from nicegui import ui
import random

# Example statistics (in real apps these come from database)
stats = {
    "total_patients": 320,
    "active_today": 12,
    "visits_this_week": 40,
    "pending_reports": 5,
}

def update_stats():
    # Random updates (only for simulation)
    stats["active_today"] += random.choice([-1, 1])
    if stats["active_today"] < 0:
        stats["active_today"] = 0

    stats["visits_this_week"] += random.choice([0, 1])
    stats["pending_reports"] += random.choice([-1, 1])
    if stats["pending_reports"] < 0:
        stats["pending_reports"] = 0

def refresh_cards():
    card1.set_text(f"Total Patients: {stats['total_patients']}")
    card2.set_text(f"Active Today: {stats['active_today']}")
    card3.set_text(f"Visits This Week: {stats['visits_this_week']}")
    card4.set_text(f"Pending Reports: {stats['pending_reports']}")

with ui.page("/"):
    ui.label("Clinic Summary Dashboard").classes("text-2xl mb-4")

    with ui.row().classes("w-full justify-between"):

        card1 = ui.card().classes("p-4 text-lg")
        card1_text = ui.label(f"Total Patients: {stats['total_patients']}")

        card2 = ui.card().classes("p-4 text-lg")
        card2_text = ui.label(f"Active Today: {stats['active_today']}")

        card3 = ui.card().classes("p-4 text-lg")
        card3_text = ui.label(f"Visits This Week: {stats['visits_this_week']}")

        card4 = ui.card().classes("p-4 text-lg")
        card4_text = ui.label(f"Pending Reports: {stats['pending_reports']}")

    # Automatically update everything
    def timer_update():
        update_stats()
        refresh_cards()

    ui.timer(interval=3.0, callback=timer_update)

ui.run()
```
